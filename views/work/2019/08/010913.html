congco 2019-08-01 dockerweb

# 项目分析

  项目数量较多,需要使用多个tomcat,并且每个tomcat下的项目数量也比较多。目前项目改动可能较为频繁，不能使用打包成docker镜像的方式，这里选择挂载数据卷。
  需要用到多个镜像服务，nginx,tomcat,MySQL,Redis等必须，所以采用docker-compose的方式。

# docker-compose

注意，使用空格，不要使用tab

version: "3"

services:
  webapp:
    image: nginx
    restart: always
    #端口映射
    ports:
      - "80:80"
      - "443:443"
    #这里注意，后面的nginx配置文件里面需要用到tomcat的名称
    links:
      - tomcat01:tomcat01
      - tomcat02:tomcat02
      - tomcat03:tomcat03
    volumes:
      - ./nginx/nginx.conf:/etc/nginx/nginx.conf
      - ./static-med:/www/static-med
    depends_on:
      - mysql
      - tomcat01
      - tomcat02
      - tomcat03
      - redis
  tomcat01:
    hostname: tomcat01
    image: tomcat:latest
    restart: always
    # 在项目中的MySQL连接需要使用domain的方式，而不是ip地址，同样，redis也是如此
    links:
      - mysql:mysql
      - redis:redis
    volumes:
      - ./webapps:/usr/local/tomcat/webapps

  tomcat02:
    hostname: tomcat01
    image: tomcat:latest
    restart: always
    links:
      - mysql:mysql
      - redis:redis
    volumes:
      - ./webapps:/usr/local/tomcat/webapps

  tomcat03:
    hostname: tomcat01
    image: tomcat:latest
    restart: always
    links:
      - mysql:mysql
      - redis:redis
    volumes:
      - ./webapps:/usr/local/tomcat/webapps

  mysql:
    hostname: mysql
    image: mysql:latest
    restart: always
    ports:
      - "3306:3306"
    volumes:
      - ./mysql/mysqldb:/var/lib/mysql
    environment:
      - MYSQL_ROOT_PASSWORD=123456 # MySQL密码
  redis:
    image: redis:alpine
    container_name: redis
    volumes:
      - ./redis:/data

# nginx.conf

user www-data;
worker_processes auto;
pid /run/nginx.pid;
include /etc/nginx/modules-enabled/*.conf;

events {
	worker_connections 768;
	# multi_accept on;
}

http {

	##
	# Basic Settings
	##

	sendfile on;
	tcp_nopush on;
	tcp_nodelay on;
	keepalive_timeout 65;
	types_hash_max_size 2048;
	# server_tokens off;
    server_names_hash_max_size 1024;
	server_names_hash_bucket_size 512;
	# server_name_in_redirect off;

	include /etc/nginx/mime.types;
	default_type application/octet-stream;

	##
	# SSL Settings
	##

	ssl_protocols TLSv1 TLSv1.1 TLSv1.2; # Dropping SSLv3, ref: POODLE
	ssl_prefer_server_ciphers on;

	##
	# Logging Settings
	##

	access_log /var/log/nginx/access.log;
	error_log /var/log/nginx/error.log;

	##
	# Gzip Settings
	##

	gzip on;
    # 负载均衡节点
	upstream backend {
    server tomcat01:8080;
    server tomcat02:8080;
    server tomcat03:8080;
    }
	#静态资源
    server {
		listen 		80;
		server_name staticmed.keeep.com;
		proxy_set_header X-Forwarded-Host $host;
        proxy_set_header X-Forwarded-Server $host;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        location / {
            root  /www/static-med;
        }
	}
    server {
        listen          80;
        server_name archiactattrbases.keeep.com;
        proxy_set_header X-Forwarded-Host $host;
        proxy_set_header X-Forwarded-Server $host;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        location / {
            proxy_pass http://backend/archiactattrbases-web/;
        }
    }
	include /etc/nginx/conf.d/*.conf;
	include /etc/nginx/sites-enabled/*;
}
